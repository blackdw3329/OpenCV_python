```python
import cv2

image = cv2.imread("lenna.bmp", cv2.IMREAD_GRAYSCALE)
if image is None:
    raise Exception("파일 읽기 오류")

h, w = image.shape
total_pixel = h * w
min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(image)
hist = [0] * 256

for i in range(h):
    for j in range(w):
        pixel = int(image[i, j])
        hist[pixel] += 1

max_count = hist[0]
max_pixel = 0

for i in range(1, 256):
    if hist[i] > max_count:
        max_count = hist[i]
        max_pixel = i

count_80 = hist[80]

print(f"영상의 전체 픽셀수 : {total_pixel}")
print(f"영상에서 픽셀값의 최소값 : {min_val}" )
print(f"영상에서 픽셀값의 최대값 : %d" % max_val)
print(f"빈도수가 가장 많은 픽셀값과 빈도수 : {max_pixel, max_count}")
print(f"픽셀값 80의 빈도수 : {count_80}")

cv2.imshow("src", image)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
