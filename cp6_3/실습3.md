```python
import numpy as np
import cv2

image = cv2.imread("lenna.bmp", cv2.IMREAD_GRAYSCALE)
if image is None:
    raise Exception("파일 읽기 오류")

def mycalcGrayHist(image):
    hist = np.zeros((256, 1), np.float32)
    h, w = image.shape[:2]

    for i in range(h):
        for j in range(w):
            pixel = int(image[i, j])
            hist[pixel, 0] += 1

    return hist

def mydrawGrayHistImage(hist):
    win_shape = (100, 256)
    hist_img = np.full(win_shape, 255, np.uint8)

    hist = hist.copy()
    cv2.normalize(hist, hist, 0, win_shape[0] - 1, cv2.NORM_MINMAX)

    for i in range(255):
        x1 = i
        y1 = win_shape[0] - 1 - int(hist[i, 0])

        x2 = i + 1
        y2 = win_shape[0] - 1 - int(hist[i + 1, 0])

        cv2.line(hist_img, (x1, y1), (x2, y2), 0, 1)

    return hist_img

hist = mycalcGrayHist(image)
hist_image = mydrawGrayHistImage(hist)

print(type(hist), hist.ndim, hist.shape, hist.dtype)

cv2.imshow("input image", image)
cv2.imshow("srcHist", hist_image)

cv2.waitKey(0)
cv2.destroyAllWindows()
```
