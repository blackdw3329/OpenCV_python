```python
import numpy as np
import cv2

image = cv2.imread("lenna.bmp", cv2.IMREAD_COLOR)
if image is None:
    raise Exception("파일 읽기 오류")

h, w = image.shape[0], image.shape[1]
gray = np.zeros((h, w), np.uint8)
dst = np.zeros((h, w), np.uint8)

for i in range(h):
    for j in range(w):
        b = int(image[i, j, 0])
        g = int(image[i, j, 1])
        r = int(image[i, j, 2])

        gray[i, j] = int(0.299 * r + 0.587 * g + 0.114 * b)

for i in range(h):
    for j in range(w):
        x = int(gray[i, j])
        if x < 128:
            y = (128 - 50) / 128 * x + 50
        elif x < 200:
            # (128,128) ~ (200,255)
            y = (255 - 128) / (200 - 128) * (x - 128) + 128
        else:
            y = 255
        if y > 255:
            y = 255
        elif y < 0:
            y = 0

        dst[i, j] = int(y)

cv2.imshow("gray", gray)
cv2.imshow("contrast", dst)

cv2.waitKey(0)
cv2.destroyAllWindows()
```
