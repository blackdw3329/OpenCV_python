```python
import numpy as np
import cv2
import time

image = cv2.imread("lenna.bmp", cv2.IMREAD_GRAYSCALE)
if image is None: raise Exception("파일 읽기 오류")

alpha = 2
beta = -128*alpha + 128

dst = np.zeros(image.shape, image.dtype)
start_time1 = time.perf_counter()
for i in range(image.shape[0]):
    for j in range(image.shape[1]):
        tmp = alpha*int(image[i,j]) + beta
        if tmp > 255: dst[i,j] = 255
        elif tmp < 0: dst[i,j] = 0
        else: dst[i,j] = tmp
elapsed = (time.perf_counter() - start_time1) * 1000
print("수행시간 : %.2f ms" % elapsed)

const = np.full(image.shape, beta, np.int32)
start_time2 = time.perf_counter()
dst2 = cv2.scaleAdd(image.astype(np.int32), alpha, const)
dst2 = dst2.clip(0,255).astype(np.uint8)
elapsed2 = (time.perf_counter() - start_time2) * 1000
print("수행시간 : %.2f ms" % elapsed2)

image_MN = cv2.minMaxLoc(image)
dst_MN = cv2.minMaxLoc(dst)
dst2_MN = cv2.minMaxLoc(dst2)

print(f"그냥 lenna 픽셀값 최대값 : {image_MN[1]}, 최소값 : {image_MN[0]}")
print(f"기본적인 명암비 조절 lenna 픽셀값 최대값 : {dst_MN[1]}, 최소값 : {dst_MN[0]}")
print(f"효율적인 명암비 조절 lenna 픽셀값 최대값 : {dst2_MN[1]}, 최소값 : {dst2_MN[0]}")

cv2.imshow("image", image)
cv2.imshow("bright", dst)
cv2.imshow("dark", dst2)
cv2.waitKey(0)
```
> 3가지 명암비를 비교했을때 어두운 픽셀은 더 어둡게 밝은 픽셀은 더 밝게 조정하여 뚜렷한 구분이 되게 만들었고 효율적인 명암비 조절이 시간이 훨씬 작았다
