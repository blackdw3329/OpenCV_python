```python
import cv2
image = cv2.imread('lenna.bmp',cv2.IMREAD_GRAYSCALE)
if image is None:
    raise Exception("파일 읽기 오류")
bright = image.copy()
dark = image.copy()

def bright_max(n):
    n = int(n)
    if((n+50)>255):
        return n
    else:
        return n + 50

def bright_min(n):
    n = int(n)
    if((n-50)<0):
        return n
    else:
        return n - 50

print(image.shape)

for row in range(image.shape[0]):
    for col in range(image.shape[1]):
        bright[row][col] = bright_max(bright[row][col])
        dark[row][col] = bright_min(dark[row][col])


cv2.imshow('Lenna Image', image)
cv2.imshow('bright Image', bright)
cv2.imshow('dark Image', dark)
cv2.waitKey(0)
cv2.destroyAllWindows()
```
